แม่ค้าเขียงหมูในตลาดสดหลายแห่ง จังหวัดหนองคาย ปรับราคาขึ้นกิโลกรัมละ 10-20 บาท อ้างอากาศหนาวหมูไม่โต 
 
จากการสำรวจตามตลาดสดหลายแห่งใน จ.หนองคาย โดยที่ตลาดสดเทศบาลเมืองท่าบ่อ และตลาดสดเทศบาลตำบลศรีเชียงใหม่ พ่อค้าแม่ค้าเขียงหมูในตลาด ต่างปรับราคาหมูขึ้นอีกกิโลกรัมละ 10-20 บาท จากราคากิโลกรัมละ 140 บาท ทั้งหมูเนื้อแดง สามชั้น สันนอก ส่วนหมูสันในราคากิโลกรัมละ 160 บาท เนื่องจากราคาหมูหน้าฟาร์มปรับราคาสูงขึ้น ส่งผลให้พ่อค้าแม่ค้าต้องปรับราคาขายตามไปด้วย
โดยแม่ค้าเขียงหมูรายหนึ่ง กล่าวว่า ราคาหมูหน้าฟาร์มได้มีการปรับขึ้น โดยราคาอยู่ที่กิโลกรัมละ 82 บาท จากเมื่อก่อน 77 บาท โดยเจ้าของฟาร์มอ้างว่า หัวอาหารที่ใช้เลี้ยงหมูมีราคาสูง อีกทั้งอยู่ในช่วงฤดูหนาว หมูโตช้า ไม่ค่อยกินอาหาร ทำให้น้ำหนักไม่ถึงเกณฑ์ จึงต้องมีการปรับขึ้นราคาหน้าฟาร์ม ทำให้ราคาขายหน้าเขียงต้องปรับขึ้นตามไปด้วย
ทั้งนี้ จากการสำรวจยังพบว่า อาหารทะเลมีการปรับขึ้นราคาทุกชนิด โดยเฉพาะ กุ้งขาว ขนาด 40 ตัว 1 กิโลกรัม จากเดิม 320 บาท ปรับขึ้นเป็น 400 บาท ขนาด 70 ตัว จาก 280 บาท ปรับขึ้นเป็น 360 บาท แต่ก็ยังมีประชาชนออกมาเลือกซื้อกันตามปกติ
